pub enum InputEvent {
    KeyPress {
        pub key: Key,
    },
    KeyRelease {
        pub key: Key,
    },
    MousePress {
        pub button: Button,
    },
    MouseRelease {
        pub button: Button,
    },
    MouseMove {
        pub x: f64,
        pub y: f64,
        pub draw_x: f64,
        pub draw_y: f64,
    },
    MouseRelativeMove {
        pub dx: f64,
        pub dy: f64,
        pub draw_dx: f64,
        pub draw_dy: f64,
    },
    MouseScroll {
        pub dx: f64,
        pub dy: f64,
    },
}
