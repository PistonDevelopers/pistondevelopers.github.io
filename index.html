About Piston

Piston is a user friendly open source game engine written in Rust, started by Sven Nilsen.

Rectangle Example

{% highlight Rust %} use graphics::*; use piston::{ GameIterator, GameIteratorSettings, GameWindow, Input, Render, Update }; fn main() { let mut window = ...; // Initialize window. let game_iter_settings = GameIteratorSettings { updates_per_second: 60, max_frames_per_second: 60 }; let mut gl = Gl::new(); for e in GameIterator::new(&mut window, &game_iter_settings) { match e { Render(args) => { gl.viewport(0, 0, args.width as i32, args.height as i32); // Start with absolute coordintes. let ref c = Context::abs(args.width as f64, args.height as f64); // Clear background with white color. c.rgb(1.0, 1.0, 1.0).draw(gl); // Draw a red rectangle in upper left corner c.rect(0.0, 0.0, 50.0, 50.0).rgb(1.0, 0.0, 0.0).draw(gl); } Update(args) => { ... } Input(e) => { ... } } } } {% endhighlight %}
