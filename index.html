Piston

A user friendly game engine written in Rust

About Piston

Piston is a user friendly open source game engine written in Rust, started by Sven Nilsen.

Rectangle Example

{% highlight Rust %} use graphics::*; use piston::{Game, RenderArgs}; pub struct App { ... } // your application data impl Game for App { fn render(&mut self, args: &RenderArgs) { let ref mut gl = self.gl; gl.viewport(0, 0, args.width as i32, args.height as i32); // Start with absolute coordintes. let ref c = Context::abs(args.width as f64, args.height as f64); // Clear background with white color. c.rgb(1.0, 1.0, 1.0).draw(gl); // Draw a red rectangle in upper left corner c.rect(0.0, 0.0, 50.0, 50.0).rgb(1.0, 0.0, 0.0).draw(gl); } } {% endhighlight %}
